# dependencias
pip install fastapi uvicorn requests celery redis python-dotenv

# celery necesita ser iniciado con venv
.\venv\Scripts\activate
python -m celery -A celery_worker.celery_app worker -l INFO -P solo

# necesita tener un redis andando
docker run -p 6379:6379 redis:7
